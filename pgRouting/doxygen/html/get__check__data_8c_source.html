128 void pgr_fetch_column_info(
129  Column_info_t info[],
130  int info_size) {
131  int i;
132  for (i = 0; i < info_size; ++i) {
133  if (fetch_column_info(&info[i])) {
134  switch (info[i].eType) {
135  case ANY_INTEGER:
136  pgr_check_any_integer_type(info[i]);
137  break;
138  case ANY_NUMERICAL:
139  pgr_check_any_numerical_type(info[i]);
140  break;
141  case TEXT:
142  pgr_check_text_type(info[i]);
143  break;
144  case CHAR1:
145  pgr_check_char_type(info[i]);
146  break;
147  case ANY_INTEGER_ARRAY:
148  pgr_check_any_integerarray_type(info[i]);
149  break;
150  default:
151  elog(ERROR, "Unknown type of column %s", info[i].name);
152  }
153  }
154  }
155 }

299 pgr_SPI_getChar(
300  HeapTuple *tuple, TupleDesc *tupdesc, Column_info_t info,
301  bool strict, char default_value) {
302  Datum binval;
303  bool isNull;
304  char value = default_value;
305 
306  binval = SPI_getbinval(*tuple, *tupdesc, info.colNumber, &isNull);
307  if (!(info.type == BPCHAROID)) {
308  elog(ERROR, "Unexpected Column type of %s. Expected CHAR", info.name);
309  }
310  if (!isNull) {
311  value = ((char*)binval)[1];
312  } else {
313  if (strict) {
314  elog(ERROR, "Unexpected Null value in column %s", info.name);
315  }
316  value = default_value;
317  }
318  return value;
319 }
