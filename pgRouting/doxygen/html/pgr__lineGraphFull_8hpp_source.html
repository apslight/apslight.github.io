223  void apply_transformation(

225  V_i vertexIt, vertexEnd;
226  EO_i e_outIt, e_outEnd;
227  EI_i e_inIt, e_inEnd;

229  // For every vertex in the original graph, create a line graph
230  // using the edges connected to that vertex
231  for (boost::tie(vertexIt, vertexEnd) = boost::vertices(digraph.graph);
232  vertexIt != vertexEnd; vertexIt++) {
233  V vertex = *vertexIt;
234  auto vertex_id = digraph[vertex].id;
235  for (boost::tie(e_outIt, e_outEnd) =
236  boost::out_edges(vertex, digraph.graph);
237  e_outIt != e_outEnd; e_outIt++) {
238  auto out_edge_id = digraph.graph[*e_outIt].id;
239  insert_vertex(vertex_id, out_edge_id);
240  }
241 
242  for (boost::tie(e_inIt, e_inEnd) =
243  boost::in_edges(vertex, digraph.graph);
244  e_inIt != e_inEnd; e_inIt++) {
245  auto in_edge_id = digraph.graph[*e_inIt].id;
246  insert_vertex(vertex_id, in_edge_id);
247

249  boost::out_edges(vertex, digraph.graph);
